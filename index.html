ShellSweeper APT Repository

Install

curl -fsSL https://srblank.github.io/shellsweeper/key.gpg | sudo gpg --dearmor -o /usr/share/keyrings/shellsweeper.gpg
echo "deb [signed-by=/usr/share/keyrings/shellsweeper.gpg] https://srblank.github.io/shellsweeper stable main" | sudo tee /etc/apt/sources.list.d/shellsweeper.list
sudo apt update
sudo apt install shellsweeper
